config switch
    option name    'switch0'
    option reset    '1'
    option enable_vlan '1'



#eth1 mikrotik
config interface wan
    option ifname    'eth0'
    option proto    'dhcp'



#eth2 mikrotik
config switch_vlan
    option device    'switch0'
    option vlan    '2'
    option ports    '0t 1'

config interface lan1
    option ifname    'eth1.2'
    option proto    'static'
    option ipaddr    '192.168.2.1'
    option netmask    '255.255.255.0'



#eth3 mikrotik
config switch_vlan
    option device    'switch0'
    option vlan    '3'
    option ports    '0t 4'

config interface lan2
    option ifname    'eth1.3'
    option proto    'static'
    option ipaddr    '192.168.3.1'
    option netmask    '255.255.255.0'



#eth4 mikrotik
config switch_vlan
    option device    'switch0'
    option vlan    '4'
    option ports    '0t 3'

config interface lan3
    option ifname    'eth1.4'
    option proto    'static'
    option ipaddr    '192.168.4.1'
    option netmask    '255.255.255.0'



#eth5 mikrotik
config switch_vlan
    option device    'switch0'
    option vlan    '5'
    option ports    '0t 2'

config interface lan4
    option ifname    'eth1.5'
    option proto    'static'
    option ipaddr    '192.168.5.1'
    option netmask    '255.255.255.0'

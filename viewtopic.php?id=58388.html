If I understand you right, you want access to LuCI from lan only.
Easiest is to modify /etc/config/uhttpd to only listen on 192.168.1.1:80 instead of 0.0.0.0:80

finally got it working

basic firewall config:

config 'rule'
    option 'src' 'Wifi_zone'
    option 'dest' 'wan'
    option 'proto' 'all'
    option 'target' 'ACCEPT'

config 'rule'
    option 'target' 'ACCEPT'
    option 'src' 'Wifi_zone'
    option 'proto' 'tcpudp'
    option 'dest_port' '53'

config 'rule'
    option 'target' 'ACCEPT'
    option 'src' 'Wifi_zone'
    option 'proto' 'udp'
    option 'dest_port' '67-68'

config 'rule'
    option 'src' 'Wifi_zone'
    option 'target' 'DROP'
    option 'proto' 'all'

config 'defaults'
    option 'syn_flood' '1'
    option 'drop_invalid' '1'
    option 'input' 'ACCEPT'
    option 'output' 'ACCEPT'
    option 'forward' 'REJECT'

config 'rule'
    option 'target' 'ACCEPT'
    option 'src' 'wan'
    option 'proto' 'all'

config 'rule'
    option 'target' 'ACCEPT'
    option 'src' 'lan'
    option 'proto' 'all'

config 'include'
    option 'path' '/etc/firewall.user'

config 'zone'
    option 'input' 'ACCEPT'
    option 'output' 'ACCEPT'
    option 'name' 'lan'
    option 'network' 'lan'
    option 'forward' 'REJECT'

config 'zone'
    option 'output' 'ACCEPT'
    option 'name' 'wan'
    option 'network' 'wan'
    option 'forward' 'REJECT'
    option 'masq' '1'
    option 'mtu_fix' '1'
    option 'input' 'ACCEPT'

config 'forwarding'
    option 'dest' 'wan'
    option 'src' 'lan'

config 'zone'

    option 'network' 'Wifi'

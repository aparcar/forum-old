Hello, I am trying to configure my raspberry pi (model B 2011.12 E1) as modem router. I have two physical Ethernet interfaces, eth0 and eth1.
To connect to my ISP I need to establish a PPPoE connection to tag traffic with vlan6.
In all the examples there is a switch configuration with multiple ports, but I only have two interfaces.

My /etc/config/networks is:


config interface 'loopback'
        option ifname 'lo'
        option proto 'static'
        option ipaddr '127.0.0.1'
        option netmask '255.0.0.0'

config interface 'lan'
        option ifname 'eth0'
        option _orig_ifname 'eth0'
        option _orig_bridge 'false'
        option proto 'dhcp'

config switch 'eth1'
        option enable '1'
        option enable_vlan '1'
        option reset '1'

config switch_vlan
        option device 'eth1'
        option vlan '6'
        option ports '0t'

config interface 'wan'
        option proto 'pppoe'
        option username 'xxxxxxxxx'
        option password 'xxxxxxx'
        option _orig_ifname 'eth1.6'
        option _orig_bridge 'false'
        option ifname 'eth1.1'

config switch_vlan
        option device 'eth1'
        option vlan '1'


I have an error:

wan (820): /usr/sbin/pppd: unrecognized option 'nic-eth1.1'

Any idea?

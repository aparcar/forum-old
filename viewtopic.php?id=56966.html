PKG_NAME:=performance
PKG_RELEASE:=1

  TITLE:=performance packet

 If you can't figure out what this program does,

 medical attention.

endef

define Build/Configure
endef

define Build/Compile
    $(MAKE) -C $(PKG_BUILD_DIR) \
    CC="$(TARGET_CC)" \
    CFLAGS="$(TARGET_CFLAGS)" \
    LDFLAGS="$(TARGET_LDFLAGS)"

Hi Cylac.

(Bell has internet on vlan35 and iptv on vlan36
I'd like to use 2 ports for IPTV (port 1 and 2) and 2 ports (port 3 and 4) for Internet (I have a switch).)

You need modify /etc/config/network look like:

config interface 'lan'
        option force_link '1'
        option type 'bridge'
        option proto 'static'
        option ipaddr '192.168.1.1'
        option netmask '255.255.255.0'
        option ip6assign '60'
        option _orig_ifname 'eth1 wlan0 wlan1'
        option _orig_bridge 'true'
        option ifname 'eth1.1'
        option igmp_snooping '1'

config interface 'wan'
        option _orig_ifname 'eth0'
        option _orig_bridge 'false'
        option ifname 'eth0.35'
        option proto 'pppoe'
        option username 'XXXXXX'
        option password 'XXXXXX'
        option peerdns '0'
        option dns '208.67.222.222 208.67.220.220' #OPENDNS SERVERS
        option delegate '0'
        option macaddr 'xx:xx:xx:xx:xx:xx' #Cloned from Bell router. Not sure if needed.

config interface 'iptv'
        option proto 'dhcp'
        option ifname 'eth0.36'
        option type 'bridge'
        option ifname 'eth0.36 eth1.2'
        option delegate '0'
        option defaultroute '0'
        option peerdns '0'

config switch
        option name 'switch0'
        option reset '1'
        option enable_vlan '1'
        option mirror_source_port '0'
        option mirror_monitor_port '0'

config switch_vlan
        option device 'switch0'
        option vlan '1'
        option vid '1'
        option ports '0t 4 5'

config switch_vlan
        option device 'switch0'
        option vlan '2'
        option ports '0t 2 3'
        option vid '2'

config switch_vlan
        option device 'switch0'
        option vlan '3'
        option vid '35'
        option ports '1t 6t'

config switch_vlan
        option device 'switch0'
        option vlan '4'
        option vid '36'
        option ports '1t 6t'

If wan&iptv interface have ip address correctly, then run following ssh command to install igmpproxy packages:

opkg update && opkg install igmpproxy kmod-bridge

Edit /etc/config/igmpproxy:

config igmpproxy
        option quickleave 1

config phyint wan
        option network iptv
        option direction upstream
        list altnet 0.0.0.0/0

config phyint lan
        option network lan
        option direction downstream
        list altnet 192.168.1.0/24        #same subnet with your LAN subnet

Add some lines to /etc/config/firewall look like:

config zone
        option name 'iptv'
        option input 'ACCEPT'
        option output 'ACCEPT'
        option forward 'ACCEPT'
        option network 'iptv'

config rule
        option src 'iptv'
        option proto 'igmp'
        option target 'ACCEPT'

config rule
        option src 'iptv'
        option proto 'udp'
        option dest 'lan'
        option dest_ip '224.0.0.0/4'
        option target 'ACCEPT'

Reboot router to see result. This configuration work for router port 1-2 to connect iptv box, port 3-4 for internet & any device can play iptv streaming.

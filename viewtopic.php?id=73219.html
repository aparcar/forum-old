Or these alternatives:

root@OpenWrt:~# head -n 4 /tmp/board.json
{
        "model": {
                "id": "netgear,r7800",
                "name": "Netgear Nighthawk X4S R7800"

root@OpenWrt:~# cat /tmp/sysinfo/board_name
netgear,r7800

root@OpenWrt:~# cat /tmp/sysinfo/model
Netgear Nighthawk X4S R7800

The problem is solved.

code snippets:
char *a = strdup ("device.new=interface");
c = uci_alloc_context ();
uci_load(c,"/root/code/uci/device" , &pkg);
if (uci_lookup_ptr (c, &p, a, true) != UCI_OK)
  {
     uci_perror (c, "XXX");   
     return 1;
  }
  uci_set(c,&p);
  uci_commit(c, &pkg, false);

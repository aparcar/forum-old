this is a config in etc/wpa_supplicant.config
ctrl_interface=/var/run/wpa_supplicant
ctrl_interface_group=0

eapol_version=1
ap_scan=2
fast_reauth=1

#IBSS/ad-hoc mode with WPA2/AES encryption
network={
    ssid="jack-mesh"
    mode=1
    scan_ssid=1
    proto=WPA
    key_mgmt=WPA-NONE
    pairwise=NONE
    group=CCMP
#   psk="sensehuge"
}

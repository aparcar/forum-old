23 Sep 2011, 10:30

Access FailSafe mode.
Transfer openwrt-ar71xx-tl-wr741nd-v1-squashfs-sysupgrade.bin to /tmp.

root@(none)# sysupgrade -v -n /tmp/openwrt-ar71xx-tl-wr741nd-v1-squashfs-sysupgrade.bin

or

root@(none)# mtd -r write /tmp/openwrt-ar71xx-tl-wr741nd-v1-squashfs-sysupgrade.bin firmware

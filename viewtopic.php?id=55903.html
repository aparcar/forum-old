cat /www/cgi-bin/wakeup 
#!/bin/sh
cat << EOT
Status: 200 OK
Vary: Accept
Content-Type: text/html; charset=UTF-8
Cache-Control: no-cache
Expires: 0

EOT
/usr/bin/etherwake -D -i "br-lan" "AA:BB:CC:DD:EE:FF"

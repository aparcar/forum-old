Try with this:

root@OpenWrt:~# uci add firewall include
root@OpenWrt:~# uci set firewall.@include[-1].path=/etc/firewall.user
root@OpenWrt:~# uci commit firewall
root@OpenWrt:~# /etc/init.d/firewall restart

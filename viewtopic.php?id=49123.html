Hi

I just bought a WR703N for a personal hobbyist project involving polling data from a server. I installed the stable Aptitude Adjustment 12.09 r36088. So far so good.

I want the WR703N to access the internet with the ethernet interface (static IP 192.168.0.165) if the ethernet cable is connected *or* with the wireless client interface (static IP 192.168.0.166) with an automatic connection to my local access point if the Ethernet cable is disconnected. This fallback (wireless to ethernet) should be automatic. I want to use static IPs to ease the access to the WR703N in case of problem.

root@OpenWrt:~# cat /etc/config/network

config interface 'loopback'
        option ifname 'lo'
        option proto 'static'
        option ipaddr '127.0.0.1'
        option netmask '255.0.0.0'

config interface 'lan'
        option ifname 'eth0'
        option proto 'static'
        option ipaddr '192.168.0.165'
        option netmask '255.255.255.0'
        option gateway '192.168.0.254'
        option broadcast '192.168.0.255'
        option dns '192.168.0.254 8.8.8.8 8.8.4.4'

config interface 'wan'
        option proto 'static'
        option ipaddr '192.168.0.166'
        option netmask '255.255.255.0'
        option gateway '192.168.0.254'
        option broadcast '192.168.0.255'
        option dns '192.168.0.254 8.8.8.8 8.8.4.4'

root@OpenWrt:~# cat /etc/config/wireless

config wifi-device 'radio0'
        option type 'mac80211'
        option macaddr '0c:82:68:f1:3b:f4'
        option hwmode '11ng'
        option htmode 'HT20'
        list ht_capab 'SHORT-GI-20'
        list ht_capab 'SHORT-GI-40'
        list ht_capab 'RX-STBC1'
        list ht_capab 'DSSS_CCK-40'
        option channel 'auto'
        option txpower '27'
        option country 'US'

config wifi-iface
        option device 'radio0'
        option ssid 'XXXX'
        option mode 'sta'
        option network 'wan'
        option encryption 'psk2'
        option key 'XXXX'

root@OpenWrt:~#

With these files, the wireless interface is connecting to the AP *only* if the ethernet cable is also connected ?! Can you help please ?

Thanks !

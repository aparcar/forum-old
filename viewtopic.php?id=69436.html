(Second Problem)

(Work on the table a few meters)

(Third Problem)
In 802.11s Mode
They Just do not see or hear any radio

Place my simple configuration

Wireless:

config wifi-device 'radio0'
option type 'mac80211'
option path 'pci0000: 00/0000: 00: 0c.0'
option htmode 'HT20'
option country 'US'
option hwmode '11a'
txpower option '15'
channel option '44'

config wifi-iface
Option device 'radio0'
network option 'lan'
option ssid 'OpenWrt'
encryption option 'none'
option mode 'adhoc'

Network:

config interface 'loopback'
    option ifname 'lo'
    option proto 'static'
    option ipaddr '127.0.0.1'
    option netmask '255.0.0.0'

config interface 'lan'
    option ifname 'eth0'
    option type 'bridge'
    option proto 'static'
    option netmask '255.255.255.0'
    option ip6assign '60'
    option ipaddr '10.189.0.140'
    option gateway '10.189.0.20'
    option dns '8.8.8.8'

config interface 'wan'
    option ifname 'eth1'
    option proto 'dhcp'

config interface 'wan6'
    option ifname 'eth1'
    option proto 'dhcpv6'

config globals 'globals'
    option ula_prefix 'fd45:6409:017d::/48'



Thanks for your help

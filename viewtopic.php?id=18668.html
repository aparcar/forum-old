Today i found an working way for suspend my usb-disk with sdparm and without any kernel patching.
I installed sdparm and adapted an script from web that works for me:

~# cat /bin/sdparm-idle
#!/bin/sh

disk=/dev/discs/disc0/generic
interval=300    #seconds

state=`grep disk_io /proc/stat`
count=$interval
up=1

while [ true ]; do
   sleep 10
   count=$(($count-10))
   newstate=`grep disk_io /proc/stat`
   if [ "$state" = "$newstate" ]; then
      if [ $count -lt 0 ]; then
         count=$interval
         if [ $up = 1 ]; then
            #echo -e "spin-down\t" `date`
            sync
            state=`grep disk_io /proc/stat`
            sdparm --command=stop $disk > /dev/null
            up=0
         fi
      fi
   else
      #echo -e "drive is up\t" `date`
      count=$interval
      state="$newstate"
      up=1
   fi
done &

Hi,

I want to use my TP Link TL-WR941nd v3.3 as dump access point.

In my network I have a working dhcp server. Wireless clients connected to a openwrt access point doesn't receive the DHCP offers of the server.

The requests from the wireless clients are received by the server, the DHCP offers are sent to the router (proofed by sniffing on a HUB next to the wireless router), but it seems that they are not bridged to the wireless systems (sniffed on an android phone).
DHCP works on the bridged LAN ports of the router with devices connected to it.

As firmware I am using a snapshot release from 6th January 2016.

/etc/config/network:

config interface 'loopback'
        option ifname 'lo'
        option proto 'static'
        option ipaddr '127.0.0.1'
        option netmask '255.0.0.0'

config interface 'eth'
        option ifname 'eth0'
        option proto 'none'

config interface 'lan'
        option ifname 'lan1 lan2 lan3 lan4'
        option type 'bridge'
        option proto 'static'
        option netmask '255.255.255.0'
        option ipaddr '10.1.100.98'
        option broadcast '10.1.100.255'
        option dns '10.1.100.18'
        option gateway '10.1.100.100'

config interface 'wan'                 
        option _orig_ifname 'wan'    
        option _orig_bridge 'false'  
        option ifname 'wan'        
        option proto 'none'

/etc/config/wireless

config wifi-device 'radio0'
        option type 'mac80211'
        option macaddr 'xx:xx:xx:xx:xx:xx'
        option hwmode '11ng'
        list ht_capab 'SHORT-GI-40'
        list ht_capab 'DSSS_CCK-40'
        option txpower '20'
        option country 'DE'
        option htmode 'HT20'
        option channel '11'

config wifi-iface
        option device 'radio0'
        option mode 'ap'
        option key 'xxx'
        option network 'lan'
        option encryption 'psk-mixed'
        option ssid 'sidname'

Does anyone have an idea, where I make a mistake?

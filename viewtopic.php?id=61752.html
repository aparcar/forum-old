Configure /etc/config/dhcp like this:

config dhcp 'lan'
    option interface 'lan'
    option start '100'
    option limit '150'
    option dhcpv6 'server'
    option ra_management '1'
    option ra 'server'
    option leasetime '24h'
    list dhcp_option '6,208.67.222.222,208.67.220.220'
    list dns '2620:0:ccc::2'
    list dns '2620:0:ccd::2'

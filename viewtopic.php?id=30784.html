This is my results of testing wds on WRT54GL with kernel 2.6:

Wireless Config con ap side:

config 'wifi-device' 'radio0'
        option 'type' 'mac80211'
        option 'macaddr' '00:1d:7e:dc:25:df'
        option 'disabled' '0'
        option 'channel' '8'

config 'wifi-iface'
        option 'device' 'radio0'
        option 'network' 'lan'
        option 'mode' 'ap'
        option 'ssid' 'UP-PVIII-I'
        option 'encryption' 'mixed-wpa-tkip-aes'
        option 'server' '1.2.3.4'
        option 'port' '1812'
        option 'key' 'thesecret'
        option 'wpa_group_rekey' '86400' 
        option wds 1
        option 'nasid' 'UP-PVIII-IX-OF

Wireless config on sta side:

config 'wifi-device' 'radio0'
        option 'type' 'mac80211'
        option 'macaddr' '00:1d:7e:dc:23:0f'
        option 'disabled' '0'
        option 'channel' '12'
        option country AR

config 'wifi-iface'
        option 'device' 'radio0'
        option 'network' 'lan'
        option 'mode' 'sta'
        option 'ssid' 'UP-PVIII-I'
        option wds      1

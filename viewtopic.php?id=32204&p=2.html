TL-W8970 works like a charm with VDSL2.
if you need Vlan for Wan try following settings ( M-net Germany, VDSL-50 )

config atm-bridge 'atm'
        option vpi '1'
        option vci '32'
        option encaps 'llc'
        option payload 'bridged'

config interface 'wan'
        option ifname 'ptm0.40'
        option proto 'pppoe'
        option ipv6 '1'
        option macaddr ‘xx:xx:xx:xx:xx’
        option username 'user'
        option password 'passwd'

config interface 'wan6'
        option ifname '@wan'
        option proto 'dhcpv6'

config vdsl 'dsl'
        option annex 'b'
        option firmware '/lib/firmware/vdsl.bin'
        option tone 'bv'
        option xfer_mode 'ptm'

The usb disk is ntfs.

/etc/config/fstab:
config mount
        option enabled '1'
        option device '/dev/sda1'
        option target '/mnt/usb'

mount | grep /dev/sda1:
/dev/sda1 on /mnt/usb type fuseblk (rw,relatime,user_id=0,group_id=0,allow_other,blksize=4096)

fyi
15 Oct 2011, 17:58

/etc/config/network

config 'interface' 'lan'
        option 'ifname' 'eth0''
        option 'type' 'bridge'
        option 'proto' 'static'
        option 'ipaddr' '192.168.1.254'
        option 'netmask' '255.255.255.0'
        option 'gateway' '192.168.1.1'
        option 'dns' '8.8.8.8 8.8.4.4

/etc/config/wireless

config 'wifi-iface'
        option 'device' 'radio0'
        option 'network' 'lan'

Modify "/etc/opkg.conf".

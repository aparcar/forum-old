wireless.radio0.txpower='10'
wireless.radio0.channel='11'
wireless.radio0.htmode='HT20'
wireless.@wifi-iface[0]=wifi-iface

wireless.@wifi-iface[0].network='lan'
wireless.@wifi-iface[0].encryption='psk2+ccmp'

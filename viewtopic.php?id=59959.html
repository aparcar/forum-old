hey,
since i switched to openwrt i cant use 40mhz on the 2,4ghz wlan, the rate stays on 144mbit. on the 5ghz network everything works fine.

my wlan config:

config wifi-device 'radio0'
        option type 'mac80211'
        option hwmode '11g'
        option path 'platform/ar934x_wmac'
        option htmode 'HT40'
        option channel '5'
        option txpower '20'
        option country 'DE'

config wifi-iface
        option device 'radio0'
        option network 'lan'
        option mode 'ap'
        option ssid 'OpenWrt'
        option encryption 'psk2'
        option key '123'

config wifi-device 'radio1'
        option type 'mac80211'
        option channel '36'
        option hwmode '11a'
        option path 'pci0000:00/0000:00:00.0'
        option disabled '0'
        option txpower '15'
        option country 'DE'
        option htmode 'HT40'

config wifi-iface
        option device 'radio1'
        option network 'lan'
        option mode 'ap'
        option encryption 'psk2'
        option key '123'
        option ssid 'OpenWrt_5GHz'

edit: a simple "option noscan '1'" fixed it.

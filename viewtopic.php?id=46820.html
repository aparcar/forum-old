Hello everybody!
I want to compile simple C++ hello world programm for open WRT. I've got SDK uploaded from here http://downloads.openwrt.org/backfire/10.03.1/ar71xx/. It successfully compiles package using C compiler and stdio.h.

But when i'm trying to use iostream and classes it compiles but fails on linking.

Source code:
#include <iostream>

int main()
{
std::cout<<"Hello World\n";
return 0;
}

Makefile:
#=====================================
# OpenWrt Makefile
#=====================================

include $(TOPDIR)/rules.mk

PKG_NAME:=mr
PKG_VERSION:=1
PKG_RELEASE:=1

PKG_BUILD_DIR:= $(BUILD_DIR)/$(PKG_NAME)

include $(INCLUDE_DIR)/package.mk


define Package/mr
    SECTION:=utils
    CATEGORY:=Utilities
    TITLE:=mr - Mouse Remote test utility
endef

define Package/mr/description
    mr - Mouse Remote test utility
endef

define Build/Prepare
    mkdir -p $(PKG_BUILD_DIR)
    $(CP) ./src/* $(PKG_BUILD_DIR)/
endef


define Build/Compile
    $(TARGET_CXX) $(TARGET_CXXFLAGS) -c -o $(PKG_BUILD_DIR)/mr.o $(PKG_BUILD_DIR)/mr.cpp
    $(TARGET_CXX) $(TARGET_LDFLAGS) -o $(PKG_BUILD_DIR)/mr $(PKG_BUILD_DIR)/mr.o
endef

define Package/mr/install
    $(INSTALL_DIR) $(1)/bin
    $(INSTALL_BIN) $(PKG_BUILD_DIR)/mr $(1)/bin/
endef

$(eval $(call BuildPackage,mr))

The problem is:

ld: warning: ld-uClibc.so.0, needed by libc.so, not found (try using -rpath or -rpath-link)
libc.so: undefined reference to `_dl_app_init_array'
libc.so: undefined reference to `_dl_app_fini_array'
libc.so: undefined reference to `_dl_loaded_modules'



But ld-uClibc.so.0 link exists in the same folder with libc.so. What am i doing wrong?

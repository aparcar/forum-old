Add this to your /etc/config/firewall

config include
    option path /etc/firewall.user

Post #3

If you prefer UCI CLI instead of using a text editor do:

root@OpenWrt:~# uci add firewall include
root@OpenWrt:~# uci set firewall.@include[0].path=/etc/firewall.user
root@OpenWrt:~# uci commit firewall
root@OpenWrt:~# /etc/init.d/firwall restart

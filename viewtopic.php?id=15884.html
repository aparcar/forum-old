I use thi script:

#!/bin/sh
g=/proc/gpio/0_out
v=12500
i=0
 
while [ $i != 100 ]
do
i=`expr $i + 1`
 
echo $i
 
echo 1 > $g
usleep $v
echo 0 > $g
usleep $v
 
done

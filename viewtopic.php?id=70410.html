/*
* function: readn
*
* read()'s n bytes from fd
* returns # of butes read, or -1 for error
*/
int readn(register int fd, register void *ptr, register int nbytes)
{

  int nleft, nread;

  nleft = nbytes;
  while (nleft > 0) {
      nread = read(fd, ptr, nleft);
      if (nread < 0)
        return nread;
      else if (nread == 0)
        break;

      nleft -= nread;
      (char*)ptr += nread;
  }
  return (nbytes - nleft);
} /* readn */

Post #2
rolf2
13 Dec 2011, 22:02

config 'rule'
    option 'target' 'ACCEPT'
    option '_name' 'vsftpd'
    option 'src' 'wan'
    option 'proto' 'tcp'
    option 'dest_port' '21'
    option 'family' 'ipv4'

Is it possible to assign specific ipv6 prefixes to a DUID ?

For example , we have on client :

config interface 'lan6'
  option ifname '@lan'
  option proto 'dhcpv6'
  option reqprefix '64'
  option reqaddress 'try'
  option clientid '0000000000000000000000000300'

On server :

config globals 'globals'
  option ula_prefix 'FD15::/48'

config interface 'lan'
  option ip6hint '10'
  option ip6assign '60'
config host
  option name 'client'
  option ip '192.168.2.2'
  option mac 'xx:xx:xx:xx:xx:xx'
  option hostid '3'
  option duid '0000000000000000000000000300'

Client receives ip address fd15:0:0:10::3 and delegated prefix :

    "ipv6-prefix-assignment": [
        {
            "address": "fd15:0:0:11::",
            "mask": 64
        },

I want this "11" to be fixed.
Say, I want it to be always "18". How do I do that ?

Try by yourself. It's not that difficult to port and package your own package for OpenWrt. If you are on Windows and you need a Linux box install some Linux in a VM like VMware player or better use andLinux and start building.

1. Build the toolchain for your target using the build-system: http://forum.openwrt.org/viewtopic.php?id=15201
2. Use an existing OpenWrt package Makefile and modify it or use the template below.

#
# Copyright (C) 2008 OpenWrt.org
#
# This is free software, licensed under the GNU General Public License v2.
# See /LICENSE for more information.
#
# $Id$

include $(TOPDIR)/rules.mk

PKG_NAME:=<name>
PKG_VERSION:=<version>
PKG_RELEASE:=1

PKG_SOURCE:=$(PKG_NAME)-$(PKG_VERSION).tar.gz
PKG_SOURCE_URL:=<download_url>
PKG_MD5SUM:=<md5sum_from_downloaded_tarball>

include $(INCLUDE_DIR)/package.mk

define Package/<pkg_name>
  SUBMENU:=<submenu>
  SECTION:=<section>
  CATEGORY:=<category>
  TITLE:=<title>
  URL:=<website>
  DEPENDS:=<dependencies>
endef

define Package/<pkg_name>/description
 My description...
endef

CONFIGURE_ARGS+= \
    --enable-shared \
    --disable-static \

define Build/Compile
    $(MAKE) -C $(PKG_BUILD_DIR) DESTDIR="$(PKG_INSTALL_DIR)" install
endef

define Package/avrdude/install
    $(INSTALL_DIR) $(1)/usr/bin
    $(CP) $(PKG_INSTALL_DIR)/usr/bin/<pkg_name> $(1)/usr/bin/
endef

$(eval $(call BuildPackage,<pkg_name>))

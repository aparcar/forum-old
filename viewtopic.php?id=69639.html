Ok, I found the solution and posted to the device specific page.

"Internet may not work under DHCP mode. To get WAN DHCP to start working properly, the "-C" option must be erased from /lib/netifd/proto/dhcp.sh."

[ "$broadcast" = 1 ] && broadcast="-B" || broadcast=
        [ -n "$clientid" ] && clientid="-x 0x3d:${clientid//:/}" || clientid="-C"
        [ -n "$iface6rd" ] && proto_export "IFACE6RD=$iface6rd"
        [ -n "$iface6rd" ] && append dhcpopts "-O 212"
        [ -n "$zone6rd" ] && proto_export "ZONE6RD=$zone6rd"
        [ -n "$zone" ] && proto_export "ZONE=$zone"
        [ "$delegate" = "0" ] && proto_export "IFACE6RD_DELEGATE=0"

I have a TP-LINK TL-WR1043ND running OpenWrt Backfire 10.03.1 / LuCI 0.10.0 Release (0.10.0)
I have 3 static IPs assigned. One to LAN client and 2 to WiFi clients.
I've noticed that once lease time comes wifi connection drops, my VPN connection on laptop goes down, etc. In a few seconds connection get back with the same ip.
Why is that happened? If i have these IPs statically fixed by MAC address - lease time should be infinite. I do not want those lost connections every 12 h.
Any ideas how to avoid that?

package 'network'

config 'interface' 'loopback'
        option 'ifname' 'lo'
        option 'proto' 'static'
        option 'ipaddr' '127.0.0.1'
        option 'netmask' '255.0.0.0'

config 'interface' 'lan'
        option 'ifname' 'eth0.1'
        option 'type' 'bridge'
        option 'proto' 'static'
        option 'ipaddr' '192.168.1.1'
        option 'netmask' '255.255.255.0'

config 'interface' 'wan'
        option 'ifname' 'eth0.2'
        option 'proto' 'dhcp'

config 'switch'
        option 'name' 'rtl8366rb'
        option 'reset' '1'
        option 'enable_vlan' '1'

config 'switch_vlan'
        option 'device' 'rtl8366rb'
        option 'vlan' '1'
        option 'ports' '1 2 3 4 5t'

config 'switch_vlan'
        option 'device' 'rtl8366rb'
        option 'vlan' '2'
        option 'ports' '0 5t'

package 'dhcp'

config 'dnsmasq'
        option 'domainneeded' '1'
        option 'boguspriv' '1'
        option 'localise_queries' '1'
        option 'rebind_protection' '1'
        option 'rebind_localhost' '1'
        option 'local' '/lan/'
        option 'domain' 'lan'
        option 'expandhosts' '1'
        option 'authoritative' '1'
        option 'readethers' '1'
        option 'leasefile' '/tmp/dhcp.leases'
        option 'resolvfile' '/tmp/resolv.conf.auto'

config 'dhcp' 'lan'
        option 'interface' 'lan'
        option 'start' '100'
        option 'limit' '150'
        option 'leasetime' '12h'

config 'dhcp' 'wan'
        option 'interface' 'wan'
        option 'ignore' '1'

config 'host'
        option 'name' 'NAME1'
        option 'mac' '00:04:4b:XX:XX:XX'
        option 'ip' '192.168.1.2'

config 'host'
        option 'name' 'NAME2
        option 'mac' '00:1f:3c:XX:XX:XX'
        option 'ip' '192.168.1.3'

config 'host'
        option 'name' 'NAME3'
        option 'mac' '00:1b:9e:XX:XX:XX'
        option 'ip' '192.168.1.4'

config 'host'
        option 'name' 'NAME4'
        option 'mac' '1c:4b:d6:XX:XX:XX'
        option 'ip' '192.168.1.5'

package 'wireless'

config 'wifi-device' 'radio0'
        option 'type' 'mac80211'
        option 'channel' '11'
        option 'macaddr' 'f4:ec:38:XX:XX:XX'
        option 'hwmode' '11ng'
        option 'htmode' 'HT20'
        list 'ht_capab' 'SHORT-GI-40'
        list 'ht_capab' 'DSSS_CCK-40'
        option 'txpower' '27'
        option 'country' 'US'

config 'wifi-iface'
        option 'device' 'radio0'
        option 'network' 'lan'
        option 'mode' 'ap'
        option 'ssid' 'MY_SSID'
        option 'encryption' 'psk2'

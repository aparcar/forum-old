Hi all.
I'm trying to add new gpio button to my device (elink m-mini). By default, it has only one "Reset" button. I've modified its .c file (mach-el-mini.c), and add new button:

#define MINI_GPIO_BTN_INT 6

static struct gpio_keys_button mini_gpio_keys[] __initdata = {
        {
                .desc           = "reset",
                .type           = EV_KEY,
                .code           = KEY_RESTART,
                .debounce_interval = MINI_KEYS_DEBOUNCE_INTERVAL,
                .gpio           = MINI_GPIO_BTN_RESET,
                .active_low     = 0,
        },
        {
                .desc           = "interruption",
                .type           = EV_KEY,
                .code           = KEY_WPS,
                .debounce_interval = MINI_KEYS_DEBOUNCE_INTERVAL,
                .gpio           = MINI_GPIO_BTN_INT,
                .active_low     = 0,
        }

};


When I've flashed this firmware, button was initialized and I can see it with "cat /sys/kernel/debug/gpio". But when I've attached a button on it - this button just copied reset button instead of creating new. I've tried to add file called "interruption" to /etc/rc.buttons/ but it's still call to reset function, and applying /etc/rc,buttons/reset script.
What have I missed? Many thanks for any help!

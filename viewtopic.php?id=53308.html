Nothing interesting, actually. Just create tagged vlan interfaces (type bridge), and add them to the ssids.

config interface 'VLAN1'
        option proto 'none'
        option ifname 'eth0.1'
        option auto '1'
        option delegate '0'
        option type 'bridge'

config interface 'VLAN2'
        option proto 'none'
        option ifname 'eth0.2'
        option auto '1'
        option delegate '0'
        option type 'bridge'
-------------------------------
config wifi-iface
        option device 'radio0'
        option mode 'ap'
        option encryption 'none'
        option ssid 'vlan1'
        option network 'VLAN1'

config wifi-iface
        option device 'radio0'
        option mode 'ap'
        option encryption 'none'
        option ssid 'vlan2'
        option network 'VLAN2'

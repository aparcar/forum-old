It's all here:

1. vi /etc/config/wireless
config wifi-device wl0
...
option macfilter '2'
option maclist ''
:wq

2. vi /etc/rc.d/S41wlmacfilter
START=47
MACFILTER=`uci get wireless.wl0.macfilter`
MACLIST=`uci get wireless.wl0.maclist`
start() {
        wlc ifname wl0 maclist "PUT YOUR MAC LIST HERE, CAPS ON, SEPARATE HEX VALUES BY ':' AND MAC ADRESSES BY SPACE"
        wlc ifname wl0 macfilter "2"
        }
stop() {
       wlc ifname wl0 maclist none
       wlc ifname wl0 macfilter 0
:wq

3. reboot

echo -e "\niptables -I FORWARD -i $(uci -P/var/state get network.wan.ifname) -s 192.168.1.1/24 -j DROP\n\niptables -I FORWARD -i $(uci -P/var/state get network.wan.ifname) -s 192.168.1.1 -j ACCEPT\n">>/etc/firewall.user

the second line is not really needed for internet access, only for accessing the gateway (ex. web gui, etc), don't know if you want that...

ssh root@192.168.1.1
cd /overlay
# Exclude delete markers unwanted/unneeded stuff
find etc lib usr -name META*>/tmp/X.lst
find etc lib usr www -type l>>/tmp/X.lst
echo htdocs>>/tmp/X.lst
# backup modified files
ls /overlay | xargs tar -X /tmp/X.lst -cvzf /tmp/backup.tar.gz -C /overlay
# make easy downloadable
ln -s /tmp/backup.tar.gz /www/config.tar.gz
# Download backup
exit
wget http://192.168.1.1/config.tar.gz
ssh root@192.168.1.1
# free RAM:
rm /tmp/X.lst /tmp/backup.tar.gz

I would prefer a history batch of uci commands done, than files whose syntax may vary between revisions.

static struct mtd_partition osbridge_parts[] = {
    { name: "bootloader", offset: 0, size: 0x10000, },

    { name: "rootfs", offset: 0, size: 0, },
    { name: "flash", offset: 0, size: 0, },
    { name: NULL, },
};

Hi,

I have a DLink DIR 615 rev E4, with Backfire 10.3 (jffs2) installed on it.

I have compiled a simple client-server program using c.

Now, I have put "utilities" as it's section. So, to run the package, everytime I have to ssh into the router and run it.

My  question is : Is there any way by which the package can start automatically when the router is switched on?

My package makefile :

include $(TOPDIR)/rules.mk
PKG_NAME:=clientserver
PKG_RELEASE:=1

PKG_BUILD_DIR := $(BUILD_DIR)/$(PKG_NAME)
include $(INCLUDE_DIR)/package.mk


define Package/clientserver
    SECTION:=utils
    CATEGORY:=Utilities
    TITLE:=clientserver -- Simple client server program
endef

define Package/clientserver/description
    This package waits for a TCP connection at port 5000. It accepts any incoming requests, and echoes back the data to the sender.
   
endef

define Build/Prepare
    mkdir -p $(PKG_BUILD_DIR)
    $(CP) ./src/* $(PKG_BUILD_DIR)/
endef

define Package/clientserver/install
    $(INSTALL_DIR) $(1)/bin
    $(INSTALL_BIN) $(PKG_BUILD_DIR)/clientserver $(1)/bin/
endef

$(eval $(call BuildPackage,clientserver))


Thanks,

Darshan

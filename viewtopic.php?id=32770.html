The buildbots do nothing fancy.

cat <<EOT > .config
CONFIG_TARGET_ar71xx=y
CONFIG_ALL=y
# CONFIG_PACKAGE_ebtables is not set
EOT
make defconfig
make IGNORE_ERRORS=m

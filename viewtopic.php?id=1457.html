#! /bin/sh
echo "Setting radio for kismet_drone"
mkdir /var/log
/sbin/ifconfig eth1 up

echo "Running kismet_drone"
/usr/bin/kismet_drone >/dev/null 2>&1 &

echo "kismet_drone now running"

i have the same setup but on a dockstar:

root@OpenWrt:~# cat /etc/config/network
config 'interface' 'loopback'
        option 'ifname' 'lo'
        option 'proto' 'static'
        option 'ipaddr' '127.0.0.1'
        option 'netmask' '255.0.0.0'

config 'interface' 'lan'
        option 'ifname' 'eth0'
        option proto 'static'
        option ipaddr '10.1.2.5'
        option netmask '255.255.255.0'

config interface 'wwan'
        option ifname 'wlan0'
        option proto 'static'
 option ipaddr '10.1.1.11'
        option netmask '255.255.255.0'
        option gateway 10.1.1.1
        option dns '8.8.8.8'
        option accept_ra '1'

modify it with your ip addresses/gateway mask accordingly

and

root@OpenWrt:~# cat /etc/config/wireless
config 'wifi-device' 'radio0'
        option 'type' 'mac80211'
        option 'phy' 'phy0'
        option 'hwmode' '11ng'
        option 'htmode' 'HT20'
        list 'ht_capab' 'SHORT-GI-40'
        list 'ht_capab' 'DSSS_CCK-40'
        option 'channel' '1'
        option 'country' 'US'
        option 'txpower' '17'

config wifi-iface
        option device   wlan0
        option network  wwan
        option mode     sta
        option ssid     mySSID
        option encryption psk2
        option key your_key

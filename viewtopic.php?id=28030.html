I've posted some details about my configuration here: https://forum.openwrt.org/viewtopic.php?id=27922. It's mainly about increasing the transmission power to 100 mW (which is allowed here), because the standard seems to be 50 mW.

Here is my config:

root@router:~# uci show wireless
wireless.radio0=wifi-device
wireless.radio0.type=mac80211
wireless.radio0.macaddr=00:00:00:00:00:00
wireless.radio0.hwmode=11ng
wireless.radio0.ht_capab=SHORT-GI-40 TX-STBC RX-STBC1 DSSS_CCK-40
wireless.radio0.disabled=0
wireless.radio0.channel=6
wireless.radio0.htmode=HT40-
wireless.radio0.country=US
wireless.radio1=wifi-device
wireless.radio1.type=mac80211
wireless.radio1.macaddr=00:00:00:00:00:00
wireless.radio1.hwmode=11na
wireless.radio1.ht_capab=SHORT-GI-40 TX-STBC RX-STBC1 DSSS_CCK-40
wireless.radio1.disabled=0
wireless.radio1.country=US
wireless.radio1.htmode=HT40-
wireless.radio1.channel=161
wireless.@wifi-iface[0]=wifi-iface
wireless.@wifi-iface[0].device=radio0
wireless.@wifi-iface[0].ssid=removed
wireless.@wifi-iface[0].network=lan
wireless.@wifi-iface[0].mode=ap
wireless.@wifi-iface[0].encryption=psk2
wireless.@wifi-iface[0].key=removed
wireless.@wifi-iface[1]=wifi-iface
wireless.@wifi-iface[1].device=radio1
wireless.@wifi-iface[1].network=lan
wireless.@wifi-iface[1].mode=ap

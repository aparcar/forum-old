root@OpenWrt:~# uci set network.wan.proto=static
root@OpenWrt:~# uci set network.wan.ipaddr=1.2.3.4
root@OpenWrt:~# uci set network.wan.netmask=255.255.255.0
root@OpenWrt:~# uci set network.wan.dns=1.2.3.4
root@OpenWrt:~# uci set network.wan.gateway=1.2.3.4
root@OpenWrt:~# uci commit network
root@OpenWrt:~# ifup wan

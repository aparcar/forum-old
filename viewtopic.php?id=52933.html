However on the client:

$ ping6 google.com

$ ip -6 route
fe80::/64 dev eth1  proto kernel  metric 256

ifconfig on the client also doesn't show an ipv6 address except the fe80 one. My suspicion is that DHCPv6 relaying is somehow broken. How can I be really sure its DHCPv6 on the uplink, and how can I find out (by not telneting onto the uplink router, can't do that) whether it supports DHCPv6-PD?

config dhcp 'lan'                  
        option interface 'lan'             
        option start '100'                       
        option limit '150'          
        option leasetime '12h'             
        option ra 'relay'                        
        option dhcpv6 'relay' 
                                           
config dhcp 'wan'                                
        option interface 'wan'  
        option ignore '1'                  
                                                 
config dhcp wan6                                 
    option dhcpv6 relay       
    option ra relay           
    option ndp relay          
    option master 1

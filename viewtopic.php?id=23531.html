Changeset 19062 (https://dev.openwrt.org/changeset?new=1 â€¦ 2Fbrcm47xx) adds basic 2.6.32 support. You just have to activate it with the patch below and then do a clean build.

Index: target/linux/brcm47xx/Makefile
===================================================================
--- target/linux/brcm47xx/Makefile      (revision 19551)
+++ target/linux/brcm47xx/Makefile      (working copy)
@@ -11,7 +11,7 @@
 BOARDNAME:=Broadcom BCM947xx/953xx
 FEATURES:=squashfs usb
 
-LINUX_VERSION:=2.6.30.10
+LINUX_VERSION:=2.6.32.7
 
 include $(INCLUDE_DIR)/target.mk
 DEFAULT_PACKAGES += wpad-mini kmod-switch kmod-diag nvram

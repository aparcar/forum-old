config 'global'
    option    anon_swap    '0'
    option    anon_mount    '0'
    option    auto_swap    '1'
    option    auto_mount    '1'
    option    delay_root    '15'
    option    check_fs    '0'

config 'mount'
    option    target    '/overlay'
    option    device    '/dev/sda1'
    option    enabled    '1'

config 'swap'
    option    uuid    '6ad2199c-2e61-4062-9182-b00962424f0a'
    option    enabled    '1'

config 'mount'
    option    target    '/mnt/data'
    option    uuid    'de64d2d4-07b0-481f-8f77-1b5a422eb2dc'
    option    enabled    '1'

Has anyone managed to get extroot working with this modem? Any help would be appreciated!

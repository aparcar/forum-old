wpa_supplicant.conf:

ctrl_interface=/var/run/wpa_supplicant
update_config=1

network={
    ssid=network1"
    scan_ssid=1
    psk=abcdef0123456789  # up to 64 digits hex
    proto=RSN
    key_mgmt=WPA-PSK
    pairwise=CCMP
    priority=10
    id_str="net1"
}

network={
    ssid="network2"
    scan_ssid=1
    psk="xYZ Ghduhd64357dhfbryyffh" # up to 63 ASCII chars in quotes
    proto=RSN
    key_mgmt=WPA-PSK
    pairwise=CCMP
    priority=10
    disabled=0
    id_str="net2"
}


network={
    ssid="known_open_network"
    scan_ssid=1
    key_mgmt=NONE
    priority=1
    disabled=0
    id_str="dhcp"
}

network={
    ssid=""                    # connect to any network
    key_mgmt=NONE
    disabled=0
    id_str="dhcp"
}

interfaces:

# permit multiple configurations
iface wlan1 inet manual
  wpa-roam /etc/wpa_supplicant/wpa_supplicant.conf

# define the WLAN networks
iface net1 inet static
  address 192.168.0.3
  netmask 255.255.255.0
  gateway 192.168.0.1
 dns-nameservers 208.67.220.220 208.67.222.222 217.0.43.193 217.0.43.1 192.168.0.1
iface net2 inet static
  address 192.168.1.6
  netmask 255.255.255.0
# network 192.168.1.0
  gateway 192.168.1.1
dns-nameservers 208.67.220.220 208.67.222.222 195.50.140.178 195.50.140.248 192.168.1.1
iface dhcp inet dhcp
auto wlan1

Have fun.

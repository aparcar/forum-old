#opkg install luci-app-openvpn .you can set you vpn config through luci.

here's my /etc/config/network

config 'interface' 'vpn'
        option 'proto' 'none'
        option 'ifname' 'tun0'

/etc/config/firewall

config 'zone'
        option 'output' 'ACCEPT'
        option 'name' 'vpn'
        option 'masq' '1'
        option 'network' 'vpn'
        option 'forward' 'REJECT'
        option 'input' 'REJECT'

config 'forwarding'
        option 'dest' 'vpn'
        option 'src' 'lan'

config 'rule'
        option 'target' 'ACCEPT'
        option 'src' 'lan'
        option 'dest' 'vpn'

config 'rule'
        option 'target' 'ACCEPT'
        option 'src' 'vpn'
        option 'dest' 'lan'

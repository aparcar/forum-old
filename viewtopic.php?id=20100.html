$ cd ~/
$ svn checkout svn://svn.openwrt.org/openwrt/branches/8.09/ ~/8.09/
$ cd ~/8.09/

$ ./scripts/feeds update packages luci
$ ./scripts/feeds install -a -p luci
$ ./scripts/feeds install openvpn

$ rm -rf .config*

$ make menuconfig

- Target System: xxx
- Target Profile: xxx

- Network
  - VPN
    - openvpn: Y

$ make world

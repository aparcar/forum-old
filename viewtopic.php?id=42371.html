I'am runing Attitude Adjustment RC-1 on tp-link wr1043nd.

I'am using p910nd printserver service at older open wrt this script worked for my hp laserjet 1020, but on Attitude Adjustment RC-1 this script don't upload firmware to printer! How can i fix this?

#!/bin/sh
set -e
FIRMWARE="/usr/lib/sihp1020.dl"
DEVICE=/dev/usb/lp0
LOGFILE=/var/log/hp

if [ "$PRODUCT" = "3f0/2b17/100" -a "$ACTION" = "add" ]; then
for i in $(seq 30); do
if [ -c $DEVICE ]; then
echo "$(date) : Sending firmware to printer…" > $LOGFILE
cat $FIRMWARE > $DEVICE
echo "$(date) : done." » $LOGFILE
exit
fi
sleep 1
done
fi

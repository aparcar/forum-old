Hi!

I have created an image for DG834PN based upon original firmware (DG834PN_V1.03.39.img) and the latest brcm63xx 12.09 Attitude Adjustment image (openwrt-DG834GT_DG834PN-squashfs-cfe.bin). It was built with mkimage.sh script:

#!/bin/sh
 
if [ "$1" = "" ] || [ "$2" = "" ] || [ "$3" = "" ]; then
echo "Usage:" $0 "DG834PN_V1.03.39.img openwrt-DG834GT_DG834PN-squashfs-cfe.bin openwrt-DG834GT_DG834PN.img"
exit
fi

echo
echo "Original Image:" $1
echo "bcm963xx_fs_kernel:" $2
echo "New Image:" $3
echo
echo "Press 'y' to continue"
 
read yn
 
if [ "$yn" = "y" ]; then
tools/makeImage $3 $1 $2
echo $3 "Created!"
else
echo "EXIT!"
fi
./mkimage.sh DG834PN_V1.03.39.img openwrt-DG834GT_DG834PN-squashfs-cfe.bin openwrt-DG834GT_DG834PN.img
